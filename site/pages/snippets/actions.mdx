})

import { client } from './viem.config'

// Resolve recipient's ENS, then send them 250 USDC.
const to = await client.ens.getAddress({
  name: 'wevm.eth',
})

const transfer = await client.token.transferSync({
  amount: { formatted: '250' },
  to,
  token: 'usdc',
})
