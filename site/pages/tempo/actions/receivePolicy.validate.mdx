# receivePolicy.validate

Checks whether a transfer or mint to a receiver is allowed by the receiver's receive policy.

## Usage

:::code-group

```ts twoslash [Decorator]
import { client } from './viem.config'

const { authorized, blockedReason } = await client.receivePolicy.validate({
  receiver: '0x70997970C51812dc3A010C7d01b50e0d17dc79C8',
  sender: '0xf39Fd6e51aad88F6F4ce6aB8827279cffFb92266',
  token: '0x20c0000000000000000000000000000000000001',
})
// @log: { authorized: true, blockedReason: 'none' }
```

```ts twoslash [Standalone]
import { Actions } from 'viem/tempo'
import { client } from './viem.config'

const { authorized, blockedReason } = await Actions.receivePolicy.validate(
  client,
  {
    receiver: '0x70997970C51812dc3A010C7d01b50e0d17dc79C8',
    sender: '0xf39Fd6e51aad88F6F4ce6aB8827279cffFb92266',
    token: '0x20c0000000000000000000000000000000000001',
  },
)
// @log: { authorized: true, blockedReason: 'none' }
```

```ts twoslash [viem.config.ts] filename="viem.config.ts"
import { Client } from 'viem/tempo'

export const client = Client.create()
```

## Return Value

`{ authorized: boolean, blockedReason: 'none' | 'tokenFilter' | 'receivePolicy' }`

Whether the transfer is authorized and, if not, why: the receiver's token policy (`'tokenFilter'`) or sender policy (`'receivePolicy'`).
