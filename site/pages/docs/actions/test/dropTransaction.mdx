# dropTransaction

Removes a transaction from the mempool.

## Usage

:::code-group

```ts twoslash [Decorator]
import { client } from './viem.config'

await client.dropTransaction({
  hash: '0xe58dceb6b20b03965bb678e27d141e151d7d4efc2334c2d6a49b9fac523f7364',
})
```

```ts twoslash [Standalone]
import { Actions } from 'viem'
import { client } from './viem.config'

await Actions.test.dropTransaction(client, {
  hash: '0xe58dceb6b20b03965bb678e27d141e151d7d4efc2334c2d6a49b9fac523f7364',
})
```

```ts twoslash [viem.config.ts] filename="viem.config.ts"
import { Client, http, testActions } from 'viem'
import { anvil } from 'viem/chains'

export const client = Client.create({
  chain: anvil,
  transport: http(),
}).extend(testActions())
```

:::

## Parameters

### hash

- **Type:** `Hex`

The hash of the transaction to drop.

```ts twoslash
import { Client, http, Actions } from 'viem'

const client = Client.create({ transport: http() })
// ---cut---
await Actions.test.dropTransaction(client, {
  hash: '0xe58dceb6b20b03965bb678e27d141e151d7d4efc2334c2d6a49b9fac523f7364', // [!code focus]
  mode: 'hardhat',
})
```

### mode

- **Type:** `'anvil' | 'hardhat' | 'ganache'`
- **Default:** `'anvil'`

The test node implementation to target.

```ts twoslash
import { Client, http, Actions } from 'viem'

const client = Client.create({ transport: http() })
// ---cut---
await Actions.test.dropTransaction(client, {
  hash: '0xe58dceb6b20b03965bb678e27d141e151d7d4efc2334c2d6a49b9fac523f7364',
  mode: 'hardhat', // [!code focus]
})
```
