# setBalance

Modifies the balance of an account.

## Usage

:::code-group

```ts twoslash [Decorator]
import { client } from './viem.config'

await client.setBalance({
  address: '0xf39Fd6e51aad88F6F4ce6aB8827279cffFb92266',
  value: 1_000_000_000_000_000_000n,
})
```

```ts twoslash [Standalone]
import { Actions } from 'viem'
import { client } from './viem.config'

await Actions.test.setBalance(client, {
  address: '0xf39Fd6e51aad88F6F4ce6aB8827279cffFb92266',
  value: 1_000_000_000_000_000_000n,
})
```

```ts twoslash [viem.config.ts] filename="viem.config.ts"
import { Client, http, testActions } from 'viem'
import { anvil } from 'viem/chains'

export const client = Client.create({
  chain: anvil,
  transport: http(),
}).extend(testActions())
```

:::

## Parameters

### address

- **Type:** `Address`

The account address.

```ts twoslash
import { Client, http, Actions } from 'viem'

const client = Client.create({ transport: http() })
// ---cut---
await Actions.test.setBalance(client, {
  address: '0xf39Fd6e51aad88F6F4ce6aB8827279cffFb92266', // [!code focus]
  value: 1_000_000_000_000_000_000n,
  mode: 'hardhat',
})
```

### value

- **Type:** `bigint`

The balance (in wei) to set.

```ts twoslash
import { Client, http, Actions } from 'viem'

const client = Client.create({ transport: http() })
// ---cut---
await Actions.test.setBalance(client, {
  address: '0xf39Fd6e51aad88F6F4ce6aB8827279cffFb92266',
  value: 1_000_000_000_000_000_000n, // [!code focus]
  mode: 'hardhat',
})
```

### mode

- **Type:** `'anvil' | 'hardhat' | 'ganache'`
- **Default:** `'anvil'`

The test node implementation to target.

```ts twoslash
import { Client, http, Actions } from 'viem'

const client = Client.create({ transport: http() })
// ---cut---
await Actions.test.setBalance(client, {
  address: '0xf39Fd6e51aad88F6F4ce6aB8827279cffFb92266',
  value: 1_000_000_000_000_000_000n,
  mode: 'hardhat', // [!code focus]
})
```
